# PBR GGX Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Remplacer Blinn-Phong par Cook-Torrance PBR (GGX + Smith + Fresnel Schlick) avec UBO de paramètres matériau, ACES filmic tone mapping et correction gamma.

**Architecture:** `PipelineConfigInfo` reçoit un champ `fragSpecializationInfo` pour passer la constante de spécialisation tone map mode. `Material` prend un `ToneMapMode` à la construction et crée le pipeline color avec la spécialisation correspondante. `MaterialInstance` possède un `Buffer` UBO (set 1, binding 3) contenant `MaterialParamsUBO` (baseColorFactor, pbrFactors, emissiveFactor). `shader.frag` est réécrit en PBR complet.

**Tech Stack:** Vulkan 1.x, GLSL 450, glslc, CMake 3.20+, C++17

                                   Material                    &material,
                                   DescriptorAllocatorGrowable &allocator,
                                   std::shared_ptr<Texture>     albedo,

    : material{material},

  auto normalInfo  = normal->descriptorInfo();
  auto ormInfo     = orm->descriptorInfo();
  auto paramsInfo  = paramsBuffer->descriptorInfo();

  writer.writeImage(0, &albedoInfo)
      .writeImage(1, &normalInfo)
      .writeImage(2, &ormInfo)

## Task 6 : Réécrire `shader.frag` avec PBR Cook-Torrance + ACES + gamma

**Files:**
- Rewrite: `shaders/shader.frag`

- [ ] **Step 1 : Réécrire shader.frag**

Remplacer le contenu complet du fichier :

```glsl
#version 450

layout (location = 0) in vec3 fragColor;
layout (location = 1) in vec3 fragPosWorld;
layout (location = 2) in vec2 fragUV;
layout (location = 3) in mat3 fragTBN;

layout (location = 0) out vec4 outColor;

layout (set = 1, binding = 0) uniform sampler2D albedoSampler;
layout (set = 1, binding = 1) uniform sampler2D normalSampler;
layout (set = 1, binding = 2) uniform sampler2D ormSampler;

layout (set = 1, binding = 3) uniform MaterialParams {
  vec4  baseColorFactor;  // multiplie albedo
  vec4  pbrFactors;       // x=metallic, y=roughness, z=normalScale, w=occlusionStrength
  vec3  emissiveFactor;
  float _pad;
} mat;

struct PointLightData {
  vec4 position;
  vec4 color;
};

layout(std430, set = 0, binding = 0) readonly buffer GlobalSceneData {
  mat4           projection;
  mat4           view;
  vec4           ambientLightColor;
  PointLightData pointLights[10];
  int            numLights;
} scene;

layout(push_constant) uniform Push {
  mat4 modelMatrix;
  mat4 normalMatrix;
} push;

layout(constant_id = 0) const uint TONE_MAP_MODE = 0;
// 0 = ACES filmic, 1 = Reinhard, 2 = Khronos PBR Neutral (futur)

const float PI = 3.14159265359;

// ─── TBN par dérivées écran ──────────────────────────────────────────────────
mat3 computeTBN(vec3 N) {
  vec3  q1  = dFdx(fragPosWorld);
  vec3  q2  = dFdy(fragPosWorld);
  vec2  st1 = dFdx(fragUV);
  vec2  st2 = dFdy(fragUV);
  float det = st1.s * st2.t - st2.s * st1.t;
  vec3  T   = normalize((q1 * st2.t - q2 * st1.t) / det);
  T         = normalize(T - dot(T, N) * N);
  vec3  B   = cross(N, T);
  return mat3(T, B, N);
}

// ─── BRDF Cook-Torrance ──────────────────────────────────────────────────────
float distributionGGX(float NdotH, float roughness) {
  float a  = roughness * roughness;
  float a2 = a * a;
  float d  = NdotH * NdotH * (a2 - 1.0) + 1.0;
  return a2 / max(PI * d * d, 1e-7);
}

float geometrySchlickGGX(float NdotX, float roughness) {
  float k = (roughness + 1.0);
  k = (k * k) / 8.0;
  return NdotX / max(NdotX * (1.0 - k) + k, 1e-7);
}

float geometrySmith(float NdotL, float NdotV, float roughness) {
  return geometrySchlickGGX(NdotL, roughness) * geometrySchlickGGX(NdotV, roughness);
}

vec3 fresnelSchlick(float cosTheta, vec3 F0) {
  return F0 + (1.0 - F0) * pow(clamp(1.0 - cosTheta, 0.0, 1.0), 5.0);
}

// ─── Tone mapping ────────────────────────────────────────────────────────────
vec3 toneMapACES(vec3 x) {
  const float a = 2.51, b = 0.03, c = 2.43, d = 0.59, e = 0.14;
  return clamp((x * (a * x + b)) / (x * (c * x + d) + e), 0.0, 1.0);
}

vec3 toneMapReinhard(vec3 x) {
  return x / (x + vec3(1.0));
}

vec3 applyToneMap(vec3 color) {
  if      (TONE_MAP_MODE == 1) return toneMapReinhard(color);
  else                         return toneMapACES(color);
}

// ─── Main ────────────────────────────────────────────────────────────────────
void main() {
  // Normale
  vec3 N_geom = normalize(fragTBN[2]);
  mat3 TBN    = computeTBN(N_geom);
  vec3 normalSample = texture(normalSampler, fragUV).rgb * 2.0 - 1.0;
  normalSample.xy  *= mat.pbrFactors.z; // normalScale
  vec3 N = normalize(TBN * normalSample);

  // ORM : R=AO, G=roughness, B=metallic
  vec3  ormSample = texture(ormSampler, fragUV).rgb;
  float ao        = 1.0 + mat.pbrFactors.w * (ormSample.r - 1.0); // occlusionStrength
  float roughness = clamp(ormSample.g * mat.pbrFactors.y, 0.04, 1.0);
  float metallic  = clamp(ormSample.b * mat.pbrFactors.x, 0.0,  1.0);

  // Albedo
  vec3 albedo = (texture(albedoSampler, fragUV).rgb * mat.baseColorFactor.rgb);

  // F0 : diélectrique = 0.04, métal = albedo
  vec3 F0 = mix(vec3(0.04), albedo, metallic);

  // Vue
  vec3 cameraPos = -(transpose(mat3(scene.view)) * scene.view[3].xyz);
  vec3 V         = normalize(cameraPos - fragPosWorld);
  float NdotV    = max(dot(N, V), 1e-4);

  // Accumulation des lumières
  vec3 Lo = vec3(0.0);
  for (int i = 0; i < scene.numLights; i++) {
    vec3  dirToLight  = scene.pointLights[i].position.xyz - fragPosWorld;
    float attenuation = 1.0 / dot(dirToLight, dirToLight);
    vec3  L           = normalize(dirToLight);
    vec3  H           = normalize(V + L);
    float NdotL       = max(dot(N, L), 0.0);
    float NdotH       = max(dot(N, H), 0.0);
    float HdotV       = max(dot(H, V), 0.0);

    vec3 radiance = scene.pointLights[i].color.xyz * scene.pointLights[i].color.w * attenuation;

    // Spéculaire Cook-Torrance
    float D  = distributionGGX(NdotH, roughness);
    float G  = geometrySmith(NdotL, NdotV, roughness);
    vec3  F  = fresnelSchlick(HdotV, F0);
    vec3  fr = (D * G * F) / max(4.0 * NdotL * NdotV, 1e-7);

    // Diffuse Lambert (les métaux n'ont pas de diffuse)
    vec3 kD = (vec3(1.0) - F) * (1.0 - metallic);
    vec3 fd = kD * albedo / PI;

    Lo += (fd + fr) * radiance * NdotL;
  }

  // Ambient + AO + émissive
  vec3 ambient = scene.ambientLightColor.xyz * scene.ambientLightColor.w * albedo * ao;
  vec3 color   = ambient + Lo + mat.emissiveFactor;

  // Tone mapping + gamma
  color = applyToneMap(color);
  color = pow(color, vec3(1.0 / 2.2));

  outColor = vec4(color, 1.0);
}
```
